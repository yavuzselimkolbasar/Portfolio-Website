from pybricks.hubs import PrimeHub
from pybricks.pupdevices import Motor
from pybricks.tools import wait
from pybricks.iodevices import XboxController
from pybricks.parameters import Button, Direction, Port

# TankDrive class to control the driving motors using a tank drive method.
class TankDrive:
    def __init__(self, left_motor, right_motor):
        self.left_motor = left_motor
        self.right_motor = right_motor

    def drive(self, left_power, right_power):
        # Set motor powers to control the robot's movement
        self.left_motor.dc(left_power)
        self.right_motor.dc(right_power)

# Arm class to control the optional arm attachment.
class Arm:
    def __init__(self, arm_motor):
        self.arm_motor = arm_motor

    def move_arm(self, down):
        # Move the arm up or down based on the 'down' parameter.
        if down:
            self.arm_motor.run_target(300, -90, wait=False)  # Move arm down
        else:
            self.arm_motor.run_target(300, 0, wait=False)    # Move arm up

# Initialize the hub and motors.
hub = PrimeHub()
left_motor = Motor(Port.C, Direction.CLOCKWISE)
right_motor = Motor(Port.D, Direction.COUNTERCLOCKWISE)
arm_motor = Motor(Port.E, Direction.CLOCKWISE)

# Create instances for tank drive and arm control.
tank_drive = TankDrive(left_motor, right_motor)
arm = Arm(arm_motor)

# Wait until the Xbox controller is successfully connected via Bluetooth.
print("Waiting for Xbox controller to connect...")
while True:
    try:
        xbox = XboxController()
        print("Xbox controller connected!")
        break
    except Exception as e:
        print(f"Controller not connected: {e}")
        wait(1000)

# Define the deadzone threshold for joystick inputs to prevent unintended movement.
DEADZONE_THRESHOLD = 80  # Adjust this value as needed

# Define a speed scaling factor to reduce overall motor power for smoother control.
SPEED_SCALE_FACTOR = 0.8  # Adjust between 0.0 and 1.0

while True:
    # Read the triggers for forward and backward movement.
    triggers = xbox.triggers()
    rt = triggers[0]  # RT accelerates forward
    lt = triggers[1]  # LT accelerates backward

    # Calculate motor powers based on triggers.
    left_power = rt - lt
    right_power = rt - lt

    # Read the left joystick for steering.
    joystick = xbox.joystick_left()
    horizontal = joystick[0]  # Horizontal axis for steering
    vertical = joystick[1]    # Vertical axis (unused directly)

    # Apply deadzone to ignore minor accidental joystick movements.
    if abs(horizontal) < DEADZONE_THRESHOLD:
        horizontal = 0
    if abs(vertical) < DEADZONE_THRESHOLD:
        vertical = 0

    # Adjust steering based on horizontal movement.
    steering_adjustment = horizontal * 0.8  # Scale turn rate for smoother steering

    # Apply steering adjustments to motor powers.
    left_power -= steering_adjustment
    right_power += steering_adjustment

    # Scale down motor power for controlled movement.
    left_power *= SPEED_SCALE_FACTOR
    right_power *= SPEED_SCALE_FACTOR

    # Command the robot to drive using the calculated power values.
    tank_drive.drive(left_power, right_power)

    # Control arm movement: press Button A to move arm down, release to move arm up.
    if Button.A in xbox.buttons.pressed():
        arm.move_arm(down=True)
    else:
        arm.move_arm(down=False)

    wait(50)

# Initialize the hub and motors for steering, driving, and arm control.
hub = PrimeHub()

steering_motor = Motor(Port.B, Direction.CLOCKWISE)
power_motor = Motor(Port.D, Direction.CLOCKWISE)
vertical_motor = Motor(Port.E, Direction.CLOCKWISE)

# Create instances of Car and Arm.
car = Car(steering_motor, power_motor)
arm = Arm(vertical_motor)

# Wait for the Xbox controller to connect via Bluetooth.
print("Waiting for Xbox controller to connect...")
while True:
    try:
        xbox = XboxController()
        print("Xbox controller connected!")
        break
    except Exception as e:
        print(f"Controller not connected: {e}")
        wait(1000)

# Initialize mode: False for Car Mode, True for Arm Mode.
arm_mode = False

while True:
    # Use the Y button to toggle between Car Mode and Arm Mode.
    if Button.Y in xbox.buttons.pressed():
        arm_mode = not arm_mode
        mode_name = "Arm Mode" if arm_mode else "Car Mode"
        print(f"Mode toggled. Current mode: {mode_name}")
        wait(300)
    
    if arm_mode:
        # Arm Mode: Control the arm using the left joystick's Y-axis.
        joystick = xbox.joystick_left()
        arm.control(joystick[1])
    else:
        # Car Mode: Control steering using left joystick's X-axis and driving with triggers.
        joystick = xbox.joystick_left()
        car.steer(joystick[0])
        trigger_values = xbox.triggers()
        drive_power = trigger_values[1] - trigger_values[0]
        car.drive_power(drive_power)
        car.check_load_and_feedback(xbox)
    
    wait(50)
